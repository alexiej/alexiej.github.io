# Performance

If you compare performance of append elements and create a new dataframe based on the list. List.append() is much faster than DataFrame.append()

# DataFrame.append()

import pandas as pd
import time


start = time.time()

df = pd.DataFrame(columns=["a","b","c","d"])
for i in range(1000):
    df.append([i+1,i+2,i+3,i+4])

end = time.time()
print('Total time:', end - start)

Total time: 1.3184735774993896

# List.append()

import pandas as pd
import time

start = time.time()

l = []
for i in range(1000):
    l.append([i+1,i+2,i+3,i+4])
df = pd.DataFrame(l,columns=["a","b","c","d"])

end = time.time()
print('Total time:', end - start)

Total time: 0.0029935836791992188
