# getcontext() - Global context

from decimal import getcontext,Decimal

getcontext().prec = 2
x = Decimal(2) / Decimal(3)
print(x)

0.67

# localcontext() - Local context

from decimal import localcontext,Decimal

with localcontext() as ctx:
    ctx.prec = 2
    x = Decimal(2) / Decimal(3)
    print('inside: x =',x)
 
y = Decimal(2) / Decimal(3)
print('outside: x =',x,'y =',y)

inside: x = 0.67
outside: x = 0.67 y = 0.6666666666666666666666666667
